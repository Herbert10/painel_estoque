# Usa uma imagem oficial do Ubuntu com suporte para Firebird
FROM ubuntu:22.04

# Define o diretório de trabalho
WORKDIR /app

# Atualiza os pacotes e instala dependências essenciais
RUN apt-get update && apt-get install -y \
    python3 \
    python3-pip \
    firebird-dev \
    firebird3.0-server \
    firebird3.0-utils \
    libfbclient2 \
    && rm -rf /var/lib/apt/lists/*

# Copia os arquivos do seu projeto para o contêiner
COPY . .

# Instala as dependências do Python
RUN pip3 install --no-cache-dir -r requirements.txt

# Expor a porta que o Railway usará
EXPOSE 5000

# Define o comando para rodar a aplicação Flask
CMD ["gunicorn", "-w", "4", "-b", "0.0.0.0:5000", "app:app"]
